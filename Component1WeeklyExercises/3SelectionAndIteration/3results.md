package weektthree;

import java.util.Scanner;

public class Reverse {

	public static void main(String[] args) {
		
		Scanner Userinput = new Scanner (System.in);
		System.out.println("Enter a sentence:");
		String sentence = Userinput.nextLine();
		
		Scanner scanword = new Scanner(sentence);
		String finalResult = "";
		
		while (scanword.hasNext()) {
			String word = scanword.next();
			String reversedWord = "";
			
			for (int i = word.length() - 1; i>=0; i--) {
				reversedWord = reversedWord + word.charAt(i);
				
			}
			
			finalResult = finalResult + reversedWord + " ";
			
		}
		
		System.out.println(finalResult.trim());
		Userinput.close();
		scanword.close();

	}

}

package weektthree;

public class PunctuationMarks {

	public static void main(String[] args) {
		String poem = "Mary had a little lamb, her fleece was as white as snow, and everywhere Mary went, the lamb was sure to go. -that was a nice poem- the end. ";
		
		 int comma = 0;
		 int fullstop = 0;
		 int dash = 0;
		 
		 for(int i = 0; i< poem.length(); i++) {
			 char symbol = poem.charAt(i);
			 
			 if (symbol ==',') {
				 comma++;
			 }
			 else if (symbol == '.') {
			     fullstop++;
			 }
			 else if (symbol == '-') {
				 dash++;
			 }
			
		 }
        System.out.println("Symbol\t| Occurences");
        System.out.println("--------------------");
        System.out.println(",\t| " + comma);
        System.out.println(".\t| " + fullstop);
        System.out.println("-\t| " + dash);
	}

}
